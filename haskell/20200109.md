myLength [] = 0
myLength (_:xs) = 1 + myLength xs

mySum :: Num a => [a] -> a
mySum [] = 0
mySum (x:xs) = x + mySum xs

myHead :: [a] -> Maybe a
myHead [] = Nothing
myHead (x:xs) = Just x

myLast [] = Nothing
myLast (x:[]) = Just x
myLast (_:xs) = myLast xs
